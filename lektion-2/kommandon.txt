# för att skapa ett nytt projekt med npm (skapar en package.json fil)
  npm init -y                 (-y innebär att vi svarar yes på alla frågor som skulle ha dykt upp)

# för att installera ett paket (dependency)
  npm i <paketnamn>           (npm i express)

# för att installera ett paket som bara används under utveckling (devDependency)
  npm i -D <paketnamn>        (npm i -D nodemon)

# för att köra ett script som du har skrivit i package.json filen
  npm run <scriptnamn>        (npm run dev)
